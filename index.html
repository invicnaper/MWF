Code Example

Param :

    #include "mwf.h"
    int main(){
        Mwu::Frame frame;
        Mw::Html html;
        Mw::Methods m;
        Mw::Function funct;
        int a ;
        frame.__init("Param testing","Description");
        if(funct.url_isset()){
                a = m.i_GET("a");
                print << "a = " << a ; 
        }else{
                print << "<b>no param</b>";
        }
        frame.__end();
        return 0;
    }

this will check if the query string exist and then will print the content of the param "a"
